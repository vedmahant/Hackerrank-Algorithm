def climbingLeaderboard(scores, alice):
    scores = list(set (scores))
    scores.sort()

    n = len(scores)
    i = 0
    result = []
    for alscore in alice:

        while i<n and scores [i]<=alscore:
            i+=1
        result.append(n - i + 1)
    return result
